def partition(A,l,r)
    x = A[l]
    while l < r
        while l < r and A[r] > x
            r = r - 1
        A[l] = A[r]
        while l < r and A[l] <= x
            l = l + 1
        A[r] = A[l]
    A[l] = x
    return l
